  (function(){
    // ======= Config =======
    const canvas = document.getElementById('game');
    const ctx = canvas.getContext('2d');
    const scoreEl = document.getElementById('score');
    const highEl = document.getElementById('high');
    const speedRange = document.getElementById('speed');
    const speedLabel = document.getElementById('speedLabel');
    const gridRange = document.getElementById('grid');
    const gridLabel = document.getElementById('gridLabel');

    const playBtn = document.getElementById('playBtn');
    const pauseBtn = document.getElementById('pauseBtn');
    const restartBtn = document.getElementById('restartBtn');
    const resetHighBtn = document.getElementById('resetHigh');

    const dpad = document.querySelector('.dpad');

    let GRID = parseInt(gridRange.value, 10);       // cells per side
    let SPEED = parseInt(speedRange.value, 10);      // 1..5

    const state = {
      running:false, paused:false, over:false,
      cell: canvas.width / GRID,
      snake:[{x:Math.floor(GRID/2), y:Math.floor(GRID/2)}],
      dir:{x:1,y:0},
      nextDir:{x:1,y:0},
      food: {x: 6, y: 6},
      grow:0,
      score:0,
      high: parseInt(localStorage.getItem('snake_high') || '0', 10),
      lastTime:0, acc:0
    };
    highEl.textContent = state.high;
    gridLabel.textContent = `${GRID}Ã—${GRID}`;
    speedLabel.textContent = `${SPEED}x`;

    // ======= Helpers =======
    function randInt(min, max){ return Math.floor(Math.random()*(max-min+1))+min; }

    function placeFood(){
      let pos;
      while(true){
        pos = { x: randInt(0, GRID-1), y: randInt(0, GRID-1) };
        if(!state.snake.some(s => s.x===pos.x && s.y===pos.y)) break;
      }
      state.food = pos;
    }

    function reset(){
      state.snake = [ {x:Math.floor(GRID/2), y:Math.floor(GRID/2)} ];
      state.dir = {x:1,y:0};
      state.nextDir = {x:1,y:0};
      state.grow = 0;
      state.score = 0;
      scoreEl.textContent = '0';
      state.over = false;
      state.acc = 0; state.lastTime = 0;
      placeFood();
      draw();
    }

    function setGrid(n){
      GRID = n; gridLabel.textContent = `${GRID}Ã—${GRID}`;
      state.cell = canvas.width / GRID;
      reset();
    }

    function setSpeed(n){
      SPEED = n; speedLabel.textContent = `${SPEED}x`;
    }

    function clamp(v,min,max){ return v < min ? min : (v > max ? max : v); }

    function draw(){
      const c = state.cell;
      ctx.clearRect(0,0,canvas.width,canvas.height);

      // grid glow (subtle)
      ctx.save();
      ctx.globalAlpha = 0.06;
      ctx.strokeStyle = '#9ca3af';
      for(let i=1;i<GRID;i++){
        const p = i*c; ctx.beginPath(); ctx.moveTo(p,0); ctx.lineTo(p,canvas.height); ctx.stroke();
        ctx.beginPath(); ctx.moveTo(0,p); ctx.lineTo(canvas.width,p); ctx.stroke();
      }
      ctx.restore();

      // food
      ctx.save();
      const g = ctx.createRadialGradient(
        (state.food.x+0.5)*c, (state.food.y+0.5)*c, 2,
        (state.food.x+0.5)*c, (state.food.y+0.5)*c, c
      );
      g.addColorStop(0, '#f43f5e');
      g.addColorStop(1, 'rgba(244,63,94,.15)');
      ctx.fillStyle = g;
      roundRect((state.food.x)*c+2,(state.food.y)*c+2,c-4,c-4,6,true,false);
      ctx.restore();

      // snake
      for(let i=0;i<state.snake.length;i++){
        const s = state.snake[i];
        const t = i/(state.snake.length-1||1);
        const color = lerpColor([34,211,238],[167,139,250], t);
        ctx.fillStyle = `rgb(${color[0]},${color[1]},${color[2]})`;
        roundRect(s.x*c+1.5, s.y*c+1.5, c-3, c-3, 8, true, false);
      }

      if(state.over){
        ctx.save();
        ctx.fillStyle = 'rgba(10,14,20,.7)';
        ctx.fillRect(0,0,canvas.width,canvas.height);
        ctx.fillStyle = '#e5e7eb';
        ctx.font = '700 28px system-ui, -apple-system, Segoe UI, Roboto';
        ctx.textAlign='center';
        ctx.fillText('Game Over', canvas.width/2, canvas.height/2 - 8);
        ctx.font = '500 16px system-ui, -apple-system, Segoe UI, Roboto';
        ctx.fillStyle = '#97a3b6';
        ctx.fillText('Press R to restart', canvas.width/2, canvas.height/2 + 18);
        ctx.restore();
      }
    }

    function roundRect(x, y, w, h, r, fill, stroke){
      if (typeof r === 'number') r = {tl:r, tr:r, br:r, bl:r};
      ctx.beginPath();
      ctx.moveTo(x + r.tl, y);
      ctx.lineTo(x + w - r.tr, y);
      ctx.quadraticCurveTo(x + w, y, x + w, y + r.tr);
      ctx.lineTo(x + w, y + h - r.br);
      ctx.quadraticCurveTo(x + w, y + h, x + w - r.br, y + h);
      ctx.lineTo(x + r.bl, y + h);
      ctx.quadraticCurveTo(x, y + h, x, y + h - r.bl);
      ctx.lineTo(x, y + r.tl);
      ctx.quadraticCurveTo(x, y, x + r.tl, y);
      ctx.closePath();
      if (fill) ctx.fill();
      if (stroke) ctx.stroke();
    }

    function lerpColor(a,b,t){ return [
      Math.round(a[0] + (b[0]-a[0])*t),
      Math.round(a[1] + (b[1]-a[1])*t),
      Math.round(a[2] + (b[2]-a[2])*t)
    ]; }

    // ======= Game Loop =======
    function step(dt){
      if(!state.running || state.paused){ draw(); return; }
      if(!state.lastTime) state.lastTime = dt; 
      const delta = dt - state.lastTime; state.lastTime = dt;
      // speed mapping: 1..5 => slower..faster
      const stepMs = [240, 190, 150, 120, 95][clamp(SPEED-1,0,4)];
      state.acc += delta;
      while(state.acc >= stepMs){
        tick();
        state.acc -= stepMs;
      }
      draw();
      requestAnimationFrame(step);
    }

    function start(){ if(state.running) return; state.running=true; state.paused=false; requestAnimationFrame(step); }
    function pause(){ state.paused = !state.paused; draw(); }

    function tick(){
      // update direction (prevent direct reversal)
      if((state.dir.x + state.nextDir.x !== 0) || (state.dir.y + state.nextDir.y !== 0)){
        state.dir = state.nextDir;
      }

      const head = { x: state.snake[0].x + state.dir.x, y: state.snake[0].y + state.dir.y };

      // wrap-around walls
      head.x = (head.x + GRID) % GRID;
      head.y = (head.y + GRID) % GRID;

      // self collision
      if(state.snake.some(s => s.x===head.x && s.y===head.y)){
        state.over = true; state.running=false; 
        if(state.score > state.high){ state.high = state.score; localStorage.setItem('snake_high', String(state.high)); }
        highEl.textContent = state.high; draw(); return;
      }

      state.snake.unshift(head);

      // food eat
      if(head.x === state.food.x && head.y === state.food.y){
        state.score += 10; scoreEl.textContent = state.score;
        state.grow += 1; placeFood();
        // tiny speed-up as reward (cap at 5x)
        if(SPEED < 5) { setSpeed(SPEED+0.05); }
      }

      if(state.grow > 0){ state.grow--; } else { state.snake.pop(); }
    }

    // ======= Input =======
    const dirs = {
      ArrowUp:{x:0,y:-1}, ArrowDown:{x:0,y:1}, ArrowLeft:{x:-1,y:0}, ArrowRight:{x:1,y:0},
      KeyW:{x:0,y:-1}, KeyS:{x:0,y:1}, KeyA:{x:-1,y:0}, KeyD:{x:1,y:0}
    };

    window.addEventListener('keydown', (e)=>{
      if(e.code in dirs){ e.preventDefault(); setDir(dirs[e.code]); }
      if(e.code === 'KeyP') { pause(); }
      if(e.code === 'KeyR') { reset(); start(); }
      if(e.code === 'Space'){ if(!state.running) start(); else pause(); }
    });

    function setDir(v){
      // prevent 180Â° reverse by checking current direction
      if((v.x + state.dir.x === 0) && (v.y + state.dir.y === 0)) return;
      state.nextDir = v;
    }

    // Touch/Click D-pad
    dpad.addEventListener('click', (e)=>{
      const btn = e.target.closest('button'); if(!btn) return;
      const map = {up:{x:0,y:-1}, down:{x:0,y:1}, left:{x:-1,y:0}, right:{x:1,y:0}};
      const v = map[btn.dataset.dir]; if(v) setDir(v);
    });

    // Basic swipe support on canvas
    let startX=0, startY=0; 
    canvas.addEventListener('touchstart', (e)=>{ const t=e.touches[0]; startX=t.clientX; startY=t.clientY; }, {passive:true});
    canvas.addEventListener('touchend', (e)=>{ 
      const t=e.changedTouches[0]; const dx=t.clientX-startX, dy=t.clientY-startY;
      if(Math.hypot(dx,dy) < 20) return; // ignore taps
      if(Math.abs(dx) > Math.abs(dy)) setDir({x: Math.sign(dx), y:0});
      else setDir({x:0, y: Math.sign(dy)});
    }, {passive:true});

    // ======= UI Bindings =======
    playBtn.addEventListener('click', ()=> start());
    pauseBtn.addEventListener('click', ()=> pause());
    restartBtn.addEventListener('click', ()=> { reset(); start(); });
    resetHighBtn.addEventListener('click', ()=>{ localStorage.removeItem('snake_high'); state.high = 0; highEl.textContent = '0'; });

    speedRange.addEventListener('input', (e)=> setSpeed(parseFloat(e.target.value)));
    gridRange.addEventListener('input', (e)=> setGrid(parseInt(e.target.value,10)));

    // ======= Boot =======
    placeFood();
    draw();

    // Auto-resize canvas to a square max width of container (keeps crispness)
    function resize(){
      const rect = canvas.getBoundingClientRect();
      const size = Math.floor(Math.min(rect.width, window.innerHeight - 160));
      const dpr = Math.min(window.devicePixelRatio || 1, 2);
      canvas.width = Math.max(360, Math.round(size * dpr));
      canvas.height = canvas.width; // square
      state.cell = canvas.width / GRID;
      draw();
    }
    window.addEventListener('resize', resize);
    setTimeout(resize, 0);
  })();
